import 'package:flutter/material.dart';
import 'package:mom_fit/router/back_widget.dart';
import 'package:video_player/video_player.dart';
import 'dart:async';
import 'package:chewie/chewie.dart';

class ExercisesPage extends StatefulWidget{
  final Map<String, dynamic> exercise;
  const ExercisesPage({super.key, required this.exercise});

  @override
  State<ExercisesPage> createState() => _ExercisesPage();
}
enum ExercisePhase { idle, runningSet, waitingRest, resting, videoRest }

class _ExercisesPage extends State<ExercisesPage> {

  VideoPlayerController? _videoController;
  ChewieController? _chewieController;
  ExercisePhase _phase = ExercisePhase.idle;

  bool _isStarted = false;
  Timer? _setTimer;
  bool isResting = false;
  int currentVideoIndex = 0;
  int currentSetIndex = 0;
  bool _isPaused = false;
  int remainingSeconds = 0;




  int _parseDuration(String duration) {
    final parts = duration.split(":").map(int.parse).toList();
    if (parts.length == 3) {
      return parts[0] * 3600 + parts[1] * 60 + parts[2];
    } else if (parts.length == 2) {
      return parts[0] * 60 + parts[1];
    }
    return 0;
  }

  @override
  void initState() {
    super.initState();
    final videos = widget.exercise["videos"] as List<dynamic>;
    final currentVideo = videos.isNotEmpty ? videos[currentVideoIndex] : null;
    final sets = currentVideo != null ? currentVideo["sets"] as List<dynamic> : [];
    _initializeVideo();
  }

  void _onStart() {
    final videos = widget.exercise["videos"] as List<dynamic>;
    if (videos.isEmpty) return;

    final currentVideo = videos[currentVideoIndex];
    final sets = currentVideo["sets"] as List<dynamic>;
    if (currentSetIndex >= sets.length) return;

    final currentSet = sets[currentSetIndex];
    final int duration = currentSet["duration_seconds"];

    setState(() {
      _phase = ExercisePhase.runningSet;   // وارد فاز اجرای ست
      remainingSeconds = duration;         // تایمر ست
      _isStarted = true;
      _isPaused = false;
    });

    _videoController?.play();

    _setTimer?.cancel();
    _setTimer = Timer.periodic(const Duration(seconds: 1), (timer) {
      if (remainingSeconds > 0) {
        setState(() {
          remainingSeconds--;
        });
      } else {
        timer.cancel();
        setState(() {
          _phase = ExercisePhase.waitingRest; // ست تموم شد → منتظر شروع استراحت
          // currentSetIndex++;
        });
      }
    });
  }


  void _onPause() {
    _videoController?.pause();
    _setTimer?.cancel();
    _isPaused = true;
    setState(() {});
  }

  void _onResume() {
    _videoController?.play();
    _isPaused = false;
    _startResumeTimer();
    setState(() {});
  }

  void _startResumeTimer() {
    _setTimer = Timer.periodic(const Duration(seconds: 1), (timer) {
      if (remainingSeconds > 0) {
        setState(() {
          remainingSeconds--;
        });
      } else {
        timer.cancel();
        if (isResting) {
          setState(() {
            isResting = false;
            currentSetIndex++;
          });
          _startSetCycle();
        } else {
          setState(() {
            isResting = true;
          });
        }
      }
    });
  }

  void _goToNextVideo() {
    final videos = widget.exercise["videos"] as List<dynamic>;
    if (currentVideoIndex < videos.length - 1) {
      currentVideoIndex++;
      _initializeVideo();
    }
  }




  void _startRest(int restSeconds) {
    setState(() {
      _phase = ExercisePhase.resting;
      remainingSeconds = restSeconds;
    });

    _setTimer?.cancel();
    _setTimer = Timer.periodic(const Duration(seconds: 1), (timer) {
      if (remainingSeconds > 0) {
        setState(() {
          remainingSeconds--;
        });
      } else {
        timer.cancel();
        setState(() {
          currentSetIndex++;
          _phase = ExercisePhase.idle;
        });
      }
    });
  }


  Future<void> _initializeVideo() async {
    final videos = widget.exercise["videos"] as List<dynamic>;
    if (videos.isEmpty) return;

    final currentVideo = videos[currentVideoIndex];
    final videoPath = currentVideo["video_file"];

    currentSetIndex = 0;
    isResting = false;

    // Dispose قبلی
    _videoController?.dispose();
    _chewieController?.dispose();

    // کنترلر جدید
    _videoController = VideoPlayerController.network(
      // 'http://10.138.209.50$videoPath',
      'https://www.w3schools.com/tags/mov_bbb.mp4',
    );

    await _videoController!.initialize();

    _chewieController = ChewieController(
      videoPlayerController: _videoController!,
      autoPlay: false,
      looping: false,
      allowFullScreen: true,
      allowMuting: true,
      aspectRatio: _videoController!.value.aspectRatio,
    );

    setState(() {});
  }

  @override
  void dispose() {
    _setTimer?.cancel();
    _videoController?.dispose();
    _chewieController?.dispose();
    super.dispose();
  }


  void _onRestart() {
    _setTimer?.cancel();
    _isPaused = false;
    isResting = false;

    setState(() {
      currentSetIndex = 0;
      _phase = ExercisePhase.idle;
      remainingSeconds = 0;
    });

    _videoController?.pause();
    _videoController?.seekTo(Duration.zero);
  }

  void _startSetCycle() {
    final videos = widget.exercise["videos"] as List<dynamic>;
    if (videos.isEmpty || currentVideoIndex >= videos.length) return;

    final currentVideo = videos[currentVideoIndex];
    final sets = currentVideo["sets"] as List<dynamic>;

    if (currentSetIndex >= sets.length) {
      _goToNextVideo();
      return;
    }

    final currentSet = sets[currentSetIndex];
    final int duration = currentSet["duration_seconds"];
    final int rest = currentSet["rest_seconds"];

    // شروع حرکت
    remainingSeconds = duration;
    _setTimer?.cancel();
    _setTimer = Timer.periodic(const Duration(seconds: 1), (timer) {
      if (remainingSeconds > 0) {
        setState(() {
          remainingSeconds--;
        });
      } else {
        timer.cancel();
        setState(() {
          isResting = true;
          remainingSeconds = rest;
        });

        // تایمر استراحت
        _setTimer = Timer.periodic(const Duration(seconds: 1), (restTimer) {
          if (remainingSeconds > 0) {
            setState(() {
              remainingSeconds--;
            });
          } else {
            restTimer.cancel();
            setState(() {
              isResting = false;
              currentSetIndex++;
            });
            _startSetCycle();
          }
        });
      }
    });
  }

  Widget _buildQuarterButton(String text, {bool enabled = true}) {
    return Container(
      padding: const EdgeInsets.symmetric(horizontal: 17, vertical: 15),
      decoration: BoxDecoration(
        color: enabled ? Colors.white : const Color(0xFFEDEDED),
        borderRadius: BorderRadius.circular(8),
        border: enabled
            ? Border.all(color: const Color(0xFFF2E3E4), width: 1)
            : null,
        boxShadow: enabled
            ? [
          BoxShadow(
            color: const Color(0xFF8B9EB8).withOpacity(0.2),
            blurRadius: 4,
            offset: const Offset(0, 2),
          ),
        ]
            : null,
      ),
      child: Text(
        text,
        style: TextStyle(
          fontSize: 12,
          color: enabled ? const Color(0xFFBE5258) : const Color(0xFFADADAD),
          fontWeight: FontWeight.bold,
        ),
      ),
    );
  }
